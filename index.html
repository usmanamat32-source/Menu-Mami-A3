<script>
let harga = {
  geprek: 15000,
  lalapan: 15000,
  bakar: 17000,
  mieayam: 12000,
  miecakalang: 13000,
  miebungkus: 10000,
  bakso: 10000,
  nasgor: 15000
};

let jumlah = {
  geprek: 0,
  lalapan: 0,
  bakar: 0,
  mieayam: 0,
  miecakalang: 0,
  miebungkus: 0,
  bakso: 0,
  nasgor: 0
};

function ubah(menu, nilai) {
  jumlah[menu] += nilai;
  if (jumlah[menu] < 0) jumlah[menu] = 0;

  document.getElementById(menu).innerText = jumlah[menu];
  hitung();
}

function hitung() {
  let total = 0;
  for (let m in jumlah) {
    total += jumlah[m] * harga[m];
  }
  document.getElementById("total").innerText = total;
}

function pesan() {
  let nama = document.getElementById("nama").value;
  let text = "Halo, saya " + nama + "%0A";
  text += "Pesanan:%0A";

  for (let m in jumlah) {
    if (jumlah[m] > 0) {
      text += "- " + m + " x" + jumlah[m] + "%0A";
    }
  }

  text += "Total: Rp " + document.getElementById("total").innerText;
  text += "%0A(Harga delivery tergantung jarak)";

  let url = "https://wa.me/628XXXXXXXXXX?text=" + text;

  window.open(url);
}
